<?php
// Start the session
session_start();

// Check if the session variable is set
if(isset($_SESSION['count'])) {
    $_SESSION['count'] = $_SESSION['count'] + 1;
} else {
    $_SESSION['count'] = 1;
}

// Display the number of visits
echo "You have visited this page " . $_SESSION['count'] . " times in this session.";
?>
