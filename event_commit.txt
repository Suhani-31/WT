CREATE TABLE Event (
    eno INT PRIMARY KEY,
    title VARCHAR(50),
    date DATE
);

CREATE TABLE Committee (
    cno INT PRIMARY KEY,
    name VARCHAR(50),
    head VARCHAR(50),
    from_time TIME,
    to_time TIME,
    status VARCHAR(20)
);

CREATE TABLE Event_Committee (
    eno INT,
    cno INT,
    PRIMARY KEY (eno, cno),
    FOREIGN KEY (eno) REFERENCES Event(eno),
    FOREIGN KEY (cno) REFERENCES Committee(cno)
);

<?php
$conn = pg_connect("host=localhost dbname=test user=postgres password=yourpassword");

if(!$conn)
{
    die("Connection Failed");
}

if(isset($_POST['submit']))
{
    $title = $_POST['title'];

    // PostgreSQL UPDATE (no JOIN directly)
    $query = "
    UPDATE Committee
    SET status = 'Working'
    WHERE cno IN (
        SELECT ec.cno
        FROM Event_Committee ec
        JOIN Event e ON e.eno = ec.eno
        WHERE e.title = '$title'
    )
    ";

    if(pg_query($conn, $query))
    {
        echo "Status updated successfully!";
    }
    else
    {
        echo "Error updating status";
    }
}
?>
